## Authorize

Get OAuth authorization URL for a toolset.<br><br>
Returns a URL that the user should visit to authorize the toolset.


### Example Usage

<!-- UsageSnippet language="go" operationID="getToolsetOAuthUrl" method="get" path="/toolsets/{toolsetId}/oauth/authorize" -->
```go

import(
	"context"
	"os"
	"github.com/pipeshub-ai/pipeshub-sdk-go/models/components"
	pipeshub "github.com/pipeshub-ai/pipeshub-sdk-go"
	"log"
)

func main() {
    ctx := context.Background()

    s := pipeshub.New(
        pipeshub.WithSecurity(components.Security{
            BearerAuth: pipeshub.Pointer(os.Getenv("PIPESHUB_BEARER_AUTH")),
        }),
    )

    res, err := s.ToolsetOAuth.Authorize(ctx, "<id>", nil)
    if err != nil {
        log.Fatal(err)
    }
    if res.Object != nil {
        // handle response
    }
}
```
