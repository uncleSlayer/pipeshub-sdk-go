## SetAzureAdAuthConfig

<!-- UsageSnippet language="go" operationID="setAzureAdAuthConfig" method="post" path="/configurationManager/authConfig/azureAd" -->

	pipeshub "github.com/pipeshub-ai/pipeshub-sdk-go"
	"log"
)

func main() {
    ctx := context.Background()

    s := pipeshub.New(
        pipeshub.WithSecurity(components.Security{
            BearerAuth: pipeshub.Pointer(os.Getenv("PIPESHUB_BEARER_AUTH")),
        }),
    )

    res, err := s.AuthenticationConfiguration.SetAzureAdAuthConfig(ctx, components.AzureAdAuthConfig{
        ClientID: pipeshub.Pointer("12345678-1234-1234-1234-123456789abc"),
